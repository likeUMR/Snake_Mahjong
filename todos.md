    AI_STATE_WANDER_TIME: 1000, // 游走状态持续时间 (ms)
    AI_STATE_CHASE_TIME: 3000,  // 追逐状态最长时间 (ms)
    AI_STATE_ESCAPE_TIME: 3000, // 逃跑状态最长时间 (ms)
    AI_STATE_FORAGE_TIME: 2000, // 觅食状态最长时间 (ms)
